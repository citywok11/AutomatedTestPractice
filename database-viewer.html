You can also interact with the database using these curl commands:

# Get all todos for the user
curl http://localhost:5119/api/users/bf51f54a-e9a0-4492-8a3e-df3c496d89d4/todos

# Create a new todo
curl -X POST http://localhost:5119/api/todos \
  -H "Content-Type: application/json" \
  -d '{
    "title": "New Task",
    "description": "Task description",
    "userId": "bf51f54a-e9a0-4492-8a3e-df3c496d89d4"
  }'

# Get a specific todo
curl http://localhost:5119/api/todos/{TODO_ID}

# Update a todo (mark as completed)
curl -X PATCH http://localhost:5119/api/todos/{TODO_ID} \
  -H "Content-Type: application/json" \
  -d '{"isCompleted": true}'

# Delete a todo
curl -X DELETE http://localhost:5119/api/todos/{TODO_ID}
